import {

  Label,
  NumberField,

  Text,

interface NumberFieldPropsExtended extends NumberFieldProps {
  label?: string;
  description?: string;
  errorMessage?: string | ((validation: ValidationResult) => string);
  placeholder?: string;
}

export const InputNumber = ({
  label,
  description,
  errorMessage,
  placeholder,
  className,
  ...props
}: NumberFieldPropsExtended) => {
  return (
    <NumberField {...props}>

        className={twMerge(

};
