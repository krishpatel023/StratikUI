"use client";

import { useEffect, useState } from "react";
import { twMerge } from "tailwind-merge";

export const Switch = ({
  variant = "primary",
  className,
  isSelected = false,
  isDisabled = false,
  onChange,
  children,
  ...props
}) => {
  const [selected, setSelected] = useState(isSelected);

  const handleChange = () => {
    if (isDisabled) {
      return;
    }
    setSelected(!selected);
    if (onChange) {
      onChange(!selected);
    }
  };

  useEffect(() => {
    setSelected(isSelected);
  }, [isSelected]);

  return (
    <button
      className={twMerge(
        "group flex gap-4 justify-center items-center hover:cursor-pointer disabled:cursor-not-allowed text-foreground disabled:text-muted-foreground",
        className
      )}
      data-selected={selected}
      onClick={handleChange}
      disabled={isDisabled}
      aria-checked={selected}
      aria-disabled={isDisabled}
      {...props}
    >
      <>
        <div
          className={twMerge(
            " rounded-full min-w-14 max-w-14 flex justify-start items-center group-data-[selected=true]:justify-end group-focus:ring-2  group-disabled:opacity-50 transition-all duration-200 ease-linear",

## Switch Props

| Name       | Type                                                  | Default     | Description                                                                                                    |
| ---------- | ----------------------------------------------------- | ----------- | -------------------------------------------------------------------------------------------------------------- |
| variant    | `"primary" \| "accent" \| "destructive" \| "outline"` | `"primary"` | Determines the style of the switch. Options include `"primary"`, `"accent"`, `"destructive"`, and `"outline"`. |
| isDisabled | `boolean`                                             | `false`     | Disables the switch if set to `true`.                                                                          |
| isSelected | `boolean`                                             | `false`     | Indicates whether the switch is selected or not.                                                               |
| onChange   | `(value: boolean) => void`                            | -           | Callback function that is called when the switch is toggled. The new value is passed as an argument.           |
| className  | `string`                                              | -           | Additional classes to apply to the component.                                                                  |
| children   | `React.ReactNode`                                     | -           | The content to be rendered inside the switch component.                                                        |
