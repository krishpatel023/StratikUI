"use client";

import useScrollTo from "@registry/packages/hooks/useScrollTo/01/default-js/useScrollTo";
import { Button } from "react-aria-components";
import { twMerge } from "tailwind-merge";

export const ScrollToTop = ({ className }) => {
  const { isVisible, scroll } = useScrollTo({ threshold: 150 });

  const handleScrollToTop = () => {
    scroll(null);
  };

  return (
    <Button
      onPress={handleScrollToTop}
      className={twMerge(
        "p-2 absolute right-4 bottom-4 text-primary-foreground bg-primary hover:bg-secondary rounded-lg border border-outline-secondary",
        isVisible ? "" : "hidden",
        className
      )}
    >
      <UpIcon className="w-6 h-6" />
    </Button>
  );
};
